									+ "/applist?platform="
									+ $rootScope.selPlatform.toUpperCase();

$http({
								method : 'GET',
								url : surl
							}).success(function(data, status, headers, config) {
								$scope.appsj = data.records;
								$rootScope.appNames = data.records;
							}).error(function(data, status, headers, config) {
								// ... TODO : add error message to screen
								// alert(error);
							});
							
							

							
							@SuppressWarnings("unchecked")
	@GET
	@Path("/getAddonsList")
	@Produces("application/json")
	public Response getAddonsList(@QueryParam("platformname") String platform) {

		
		JSONObject jsonObj = new JSONObject();
		String result = null;
		

		JSONArray array = new JSONArray();

		try {
			PlatformGovDAO dao = PlatformGovDAO.getInstance();
			ArrayList<AddOns> addons = new ArrayList<AddOns>();
			  
			

		
			addons = dao.getAddonList(platform);
				for(AddOns addon:addons){
					
					array.add(addon.toJSON());
				}

				dao.close();
		} catch (Exception e) {
			// TODO: set the proper error code
			e.printStackTrace();
			jsonObj.put("ERROR", "Service Failure");
			return Response.status(500).entity(jsonObj).build();
		}
		jsonObj.put("records", array);
		result = "" + jsonObj;

		return Response.status(200).entity(result).build();

	}
